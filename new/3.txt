import pandas as pd
import matplotlib.pyplot as plt

# Read dataset
df = pd.read_csv(r"E:\TY\DSML_dataset\House Data.csv")

print(df.head())

# Select only numeric columns
dfa = df.select_dtypes(include="number")

# Standard deviation
print(dfa.std())

# Variance
print(dfa.var())

# Percentiles (median)
print(dfa.quantile(0.50))

# Histogram for each numeric feature
dfa.hist(figsize=(10,6))
plt.show()
