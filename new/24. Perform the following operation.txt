import pandas as pd

# Read dataset
df = pd.read_csv(r"E:\TY\DSML_dataset\Covid Vaccine Statewise.csv")

print(df.head())

# Counting unique values
print("\nUnique values:\n", df.nunique())

# Data types of each column
print("\nData types:\n", df.dtypes)

# Convert column data type
df["Total Doses Administered"] = df["Total Doses Administered"].astype(float)
print("\nUpdated Data type of Total Doses Administered:\n", df["Total Doses Administered"].dtype)

# Missing values count
print("\nMissing values:\n", df.isnull().sum())

# Filling missing values
df_filled = df.copy()

for col in df_filled.columns:
    if df_filled[col].dtype in ["int64", "float64"]:
        df_filled[col] = df_filled[col].fillna(df_filled[col].mean())
    else:
        df_filled[col] = df_filled[col].fillna(df_filled[col].mode()[0])

print("\nMissing values after filling:\n", df_filled.isnull().sum())
