import pandas as pd
import matplotlib.pyplot as plt

df = pd.read_csv(r"E:\TY\DSML_dataset\House Data.csv")

print(df.head())

# Clean price column
df["price"] = df["price"].str.replace(r"[^0-9]", "", regex=True).astype(float)

# Summary statistics
print("Mean:", df["price"].mean())
print("Median:", df["price"].median())
print("Minimum:", df["price"].min())
print("Maximum:", df["price"].max())
print("Standard Deviation:", df["price"].std())

# Check column names
print("\nColumns available:")
print(df.columns)

# Use 'district' column instead of province_name
print("\nSummary statistics grouped by district:")
print(df.groupby("district")["price"].describe())
